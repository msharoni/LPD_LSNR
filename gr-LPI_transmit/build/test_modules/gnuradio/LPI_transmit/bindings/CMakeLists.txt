########################################################################
# Check if there is C++ code at all
########################################################################
if(NOT LPI_transmit_sources)
    message(STATUS "No C++ sources... skipping python bindings")
    return()
endif(NOT LPI_transmit_sources)

########################################################################
# Check for pygccxml
########################################################################
gr_python_check_module_raw("pygccxml" "import pygccxml" PYGCCXML_FOUND)

include(GrPybind)

########################################################################
# Python Bindings
########################################################################

list(APPEND LPI_transmit_python_files python_bindings.cc)

gr_pybind_make_oot(LPI_transmit ../../.. gr::LPI_transmit "${LPI_transmit_python_files}")

# copy bindings extension for use in QA test module
add_custom_command(
    TARGET LPI_transmit_python
    POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy $<TARGET_FILE:LPI_transmit_python>
            ${PROJECT_BINARY_DIR}/test_modules/gnuradio/LPI_transmit/)

install(
    TARGETS LPI_transmit_python
    DESTINATION ${GR_PYTHON_DIR}/gnuradio/LPI_transmit
    COMPONENT pythonapi)
